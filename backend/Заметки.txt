//Зпрос для получения суммы площадей проданных за весь период
SELECT
    SUM(length * width) as sum_s
FROM
    `product_list`,
    `linoleum_width`
WHERE
    id_width = linoleum_width.id AND product_list.id_linoleum = 1;

//Запрос общая статистика по продажам
SELECT NAME
    ,
    SUM(LENGTH * width) AS S
FROM
    `orders`,
    `linoleum`,
    `product_list`,
    `linoleum_width`
WHERE
    product_list.id_width = linoleum_width.id AND product_list.id_linoleum = linoleum.id AND orders.id = id_order AND orders.delivery_date > "2022-05-01"
GROUP BY NAME
ORDER BY
    `S`
DESC
